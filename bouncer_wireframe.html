3 — YAML Contract Structure
rna-seq-schema.yaml
version: "1.0.0" # required assay_type: rna-seq data_stage: raw_counts index_column: gene_id sample_columns: from_samplesheet metadata_columns: - name: sample_id required: true - name: condition required: true - name: organism required: true - name: batch required: false output_features: - counts_matrix - sample_metadata
rna-seq-qc.yaml
version: "1.0.0" pipeline: nf-core/rnaseq data_stage: raw_counts multiqc: - metric: STAR_percent_uniquely_mapped min: 0.75 severity: hard - metric: FastQC_percent_duplicates max: 0.50 severity: hard design: - check: min_replicates_per_condition value: 3 severity: warning - check: batch_not_confounded_with_condition severity: warning
